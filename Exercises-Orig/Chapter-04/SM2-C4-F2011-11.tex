%% Title: Practice with Means

Read in the Current Population Survey wage data:
\begin{verbatim}
> w = fetchData("CPS.csv")
\end{verbatim}

\begin{enumerate}[(a)]
\item What is the grand mean of \VN{wage}?\\
  \SelectSetHoriz{9.02}{7.68,7.88,8.26,8.31,9.02,9.40,10.88}
\begin{AnswerText}
\begin{verbatim}
> mean(wage, data=w)
[1] 9.024064
\end{verbatim}
\end{AnswerText}
\item What is the group-wise mean of \VN{wage} for females? \SelectSetHoriz{7.88}{7.68,7.88,8.26,8.31,9.02,9.40,10.88}
\begin{AnswerText}
\begin{verbatim}
> mean(wage~sex, data=w)
  sex        S   N Missing
1   F 7.878857 245       0
2   M 9.994913 289       0
\end{verbatim}
\end{AnswerText}
\item What is the group-wise mean of \VN{wage} for married people? \SelectSetHoriz{9.40}{7.68,7.88,8.26,8.31,9.02,9.40,10.88}
\begin{AnswerText}
\begin{verbatim}
> mean(wage~married, data=w)
  married        S   N Missing
1 Married 9.398486 350       0
2  Single 8.311848 184       0
\end{verbatim}
\end{AnswerText}
\item What is the group-wise mean of \VN{wage} for married females? \SelectSetHoriz{7.68}{7.68,7.88,8.26,8.31,9.02,9.40,10.88}
\begin{AnswerText}
\begin{verbatim}
> mean(wage~married+sex, data=w)
  married sex         S   N Missing
1 Married   F  7.683765 162       0
2  Single   F  8.259639  83       0
3 Married   M 10.876064 188       0
4  Single   M  8.354752 101       0
\end{verbatim}
\end{AnswerText}

\end{enumerate}
